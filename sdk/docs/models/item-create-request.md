# ItemCreateRequest

Represents item create request.

## Example Usage

```typescript
import { ItemCreateRequest } from "plaky115/models";

let value: ItemCreateRequest = {
  title: "My New Item",
  parentId: 1,
  groupId: 1,
  groupTitle: "Backlog",
  fields: {
    "Status": "To do",
    "number-1": 50,
    "Description": "Test description",
  },
};
```
