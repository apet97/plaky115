# ItemResponse

Represents extendable item response.

## Example Usage

```typescript
import { ItemResponse } from "plaky115-mcp/models";

let value: ItemResponse = {
  board: 569087,
  space: {
    boards: [
      {
        defaultValues: {
          defaultAttributes: {
            "status-1": "1",
            "tag-1": [
              "1",
              "2",
            ],
          },
        },
      },
    ],
  },
  parent: 237542,
  fields: [
    {
      key: "status-1",
    },
  ],
  subitems: [
    {
      board: 201715,
      space: 869479,
      parent: 855941,
      fields: [
        {
          key: "status-1",
        },
      ],
    },
  ],
};
```
